static void bit_block_move_ultimate(char* dest_data, size_t dest_bit_offset,
                                    const char* src_data, size_t src_bit_offset,
                                    size_t num_bits) {
  if (num_bits == 0) {

  // åŒæ ·ï¼Œé‡å æ‹·è´çš„å¤„ç†å¯¹äºä¸€ä¸ªå¥å£®çš„å‡½æ•°æ˜¯å¿…éœ€çš„ã€‚
